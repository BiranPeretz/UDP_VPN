#server setup
./server 5555 0.0.0.0 tun0 &
ip addr add 10.10.0.1/24 dev tun0
ip link set tun0 up

#server routing and config
sysctl -w net.ipv4.ip_forward=1

#replace with real network interface
SRV_IF=wlo1
iptables -t nat -A POSTROUTING -s 10.10.0.0/24 -o "$SRV_IF" -j MASQUERADE
iptables -A FORWARD -i tun0 -o "$SRV_IF" -s 10.10.0.0/24 -j ACCEPT
iptables -A FORWARD -i "$SRV_IF" -o tun0 -m state --state RELATED,ESTABLISHED -j ACCEPT
